# ================================================
# Open-World Biological Daughter Puzzle Encoding
# ================================================

Person = DeclareSort("Person")

# Core predicates
Female   = Function("Female",   Person, BoolSort())
ParentOf = Function("ParentOf", Person, Person, BoolSort())

# Biological definition of Daughter (global, persistent)
# Daughter(x, y) means: x is a female offspring of parent y
Daughter = Function("Daughter", Person, Person, BoolSort())

ForAll([x, y],
    Daughter(x, y) == And(Female(x), ParentOf(y, x))
)

# ================================================
# Puzzle constraints - NO universe closure
# ================================================

# There exist exactly two parents: one male, one female
p_father, p_mother = Consts("p_father p_mother", Person)

Exists([p_father, p_mother],
    And(
        p_father != p_mother,
        Not(Female(p_father)),   # male parent
        Female(p_mother),        # female parent
        
        # They are parents in this family
        # (we do NOT say these are the only parents in the world)
    )
)

# There exist exactly two children who share both parents
c1, c2 = Consts("c1 c2", Person)

Exists([c1, c2],
    And(
        c1 != c2,
        
        # Both children share both parents
        ParentOf(p_father, c1),
        ParentOf(p_mother, c1),
        ParentOf(p_father, c2),
        ParentOf(p_mother, c2),
        
        # One of them is a girl and sister of the other
        Or(
            And(Female(c1), Not(Female(c2))),   # c1 girl, c2 boy
            And(Female(c2), Not(Female(c1))),   # c2 girl, c1 boy
            And(Female(c1), Female(c2))         # both girls
        )
    )
)

# ================================================
# Open World - Explicitly NO closure
# ================================================

# Important: We do NOT add any axiom like:
#   ForAll(x, Or(x == p_father, x == p_mother, x == c1, x == c2))
# This would be illegal universe closure.

# The domain may contain arbitrarily many other people (ancestors, etc.)

# ================================================
# Key Insight: Mother is a daughter
# ================================================

# Because we use the global biological definition,
# and the mother is Female, she must have parent(s) somewhere
# (this is allowed in open world - we do not forbid it)

# Assert that the mother is a daughter of someone
Exists([grandparent],
    ParentOf(grandparent, p_mother)
)

# ================================================
# Possible number of daughters in the family
# ================================================

# Daughters = all x such that Daughter(x, some y) where the relation touches the family

# Guaranteed:
# - p_mother is a daughter
# - at least one of {c1, c2} is a daughter

# Therefore minimum = 2

# If both c1 and c2 are female, then total = 3

# This directly counters the "local relational structure only" reading:
# The biological definition is global and persistent.
# The puzzle never says the mother has no parents.
# Open world allows her ancestry without violating any constraint.
